     import Vapor

     enum LaunchController {
         /// Single active launch guard — Wine on macOS doesn't handle parallel processes well
         private static let launchLock = NSLock()
         private static var activeLaunch: String? = nil

         static func register(_ app: Application) throws {
             // GET /games/:gameId/launch — launch log page
             app.get("games", ":gameId", "launch") { req async throws -> View in
                 guard let gameId = req.parameters.get("gameId") else {
                     throw Abort(.badRequest)
                 }
                 guard let game = try await GameService.shared.findGame(id: gameId) else {
                     throw Abort(.notFound)
                 }
                 let useAgent = (req.query[String.self, at: "agent"] ?? "false") == "true"
                 return try await req.view.render("launch-log", LaunchLogContext(
                     gameId: game.id,
                     gameName: game.name,
                     useAgent: useAgent,
                     streamURL: "/games/\(game.id)/launch/stream?agent=\(useAgent)"
                 ))
             }

             // GET /games/:gameId/launch/stream — SSE stream
             app.get("games", ":gameId", "launch", "stream") { req async throws -> Response in
                 guard let gameId = req.parameters.get("gameId") else {
                     throw Abort(.badRequest)
                 }

                 // Enforce single active launch
                 launchLock.lock()
                 if let active = activeLaunch {
                     launchLock.unlock()
                     throw Abort(.conflict, reason: "Game '\(active)' is currently launching")
                 }
                 activeLaunch = gameId
                 launchLock.unlock()

                 let useAgent = (req.query[String.self, at: "agent"] ?? "false") == "true"

                 let body = Response.Body(stream: { writer in
                     Task.detached {
                         defer {
                             launchLock.lock()
                             activeLaunch = nil
                             launchLock.unlock()
                         }

                         do {
                             try await runLaunch(
                                 gameId: gameId,
                                 useAgent: useAgent,
                                 writer: writer
                             )
                         } catch {
                             await sendSSE(writer: writer, event: "error",
                                          data: "<div class='error'>Error: \(error.localizedDescription)</div>")
                         }

                         // Signal completion
                         await sendSSE(writer: writer, event: "complete",
                                      data: "<div>Launch finished.</div>")
                         _ = try? await writer.write(.end).get()
                     }
                 })

                 let response = Response(status: .ok, body: body)
                 response.headers.replaceOrAdd(name: .contentType, value: "text/event-stream")
                 response.headers.replaceOrAdd(name: .cacheControl, value: "no-cache")
                 response.headers.replaceOrAdd(name: .connection, value: "keep-alive")
                 return response
             }
         }

         // MARK: - Launch Logic

         private static func runLaunch(gameId: String, useAgent: Bool, writer: Response.Body.StreamWriter) async throws {
             guard let game = try await GameService.shared.findGame(id: gameId) else {
                 throw Abort(.notFound)
             }
             guard let wineURL = LaunchService.resolveWine() else {
                 await sendSSE(writer: writer, event: "error",
                              data: "<div class='error'>Wine is not installed</div>")
                 return
             }

             let bottleURL = CellarPaths.bottleDir(for: gameId)
             guard let executablePath = game.executablePath else {
                 await sendSSE(writer: writer, event: "error",
                              data: "<div class='error'>No executable path configured for this game</div>")
                 return
             }

             await sendSSE(writer: writer, event: "status",
                          data: "<div>Preparing to launch \(game.name)...</div>")

             if useAgent {
                 try await runAgentLaunch(
                     gameId: gameId, game: game, executablePath: executablePath,
                     wineURL: wineURL, bottleURL: bottleURL, writer: writer
                 )
             } else {
                 try await runDirectLaunch(
                     gameId: gameId, game: game, executablePath: executablePath,
                     wineURL: wineURL, bottleURL: bottleURL, writer: writer
                 )
             }
         }

         private static func runDirectLaunch(
             gameId: String, game: GameEntry, executablePath: String,
             wineURL: URL, bottleURL: URL, writer: Response.Body.StreamWriter
         ) async throws {
             await sendSSE(writer: writer, event: "status",
                          data: "<div>Direct launch — applying recipe...</div>")

             // Load and apply recipe
             if let recipe = try? RecipeEngine.findBundledRecipe(for: gameId) {
                 let wineProcess = WineProcess(wineBinary: wineURL, winePrefix: bottleURL)
                 RecipeEngine.apply(recipe: recipe, using: wineProcess)
                 await sendSSE(writer: writer, event: "log",
                              data: "<div>Recipe applied: \(recipe.name ?? gameId)</div>")
             }

             // Check for user recipe
             let userRecipe = CellarPaths.userRecipeFile(for: gameId)
             if FileManager.default.fileExists(atPath: userRecipe.path),
                let data = try? Data(contentsOf: userRecipe),
                let recipe = try? JSONDecoder().decode(Recipe.self, from: data) {
                 let wineProcess = WineProcess(wineBinary: wineURL, winePrefix: bottleURL)
                 RecipeEngine.apply(recipe: recipe, using: wineProcess)
                 await sendSSE(writer: writer, event: "log",
                              data: "<div>User recipe applied</div>")
             }

             await sendSSE(writer: writer, event: "status",
                          data: "<div>Launching game...</div>")

             // Run the game
             // CRITICAL: Must run on a separate dispatch queue, NOT on NIO event loop.
             // WineProcess uses synchronous Process APIs + readabilityHandler.
             let wineProcess = WineProcess(wineBinary: wineURL, winePrefix: bottleURL)

             // Run Wine process synchronously on a detached thread and stream output
             // The readabilityHandler on WineProcess streams stderr/stdout line-by-line.
             // To wire this to SSE: create a custom run that passes each line to the writer.
             //
             // NOTE: The actual wiring depends on WineProcess API. If run() is blocking and
             // returns WineResult, we need to set up the readabilityHandler BEFORE calling run.
             // Check WineProcess implementation:
             // - If it exposes a way to set output handlers, use that
             // - If run() handles everything internally, we may need to add a lineHandler parameter
             //
             // Simplest approach for Phase 12: run the game, stream the result after completion.
             // Full line-by-line streaming can be enhanced later.
             let result = wineProcess.run(
                 executable: executablePath,
                 arguments: [],
                 environment: [:],
                 logFile: CellarPaths.logFile(for: gameId, timestamp: Date())
             )

             // Stream the result
             let statusText = result.exitCode == 0 ? "Game exited normally" : "Game exited with code \(result.exitCode)"
             await sendSSE(writer: writer, event: "status",
                          data: "<div>\(statusText)</div>")

             if !result.output.isEmpty {
                 let escaped = result.output
                     .replacingOccurrences(of: "&", with: "&amp;")
                     .replacingOccurrences(of: "<", with: "&lt;")
                     .replacingOccurrences(of: ">", with: "&gt;")
                 for line in escaped.components(separatedBy: "\n").prefix(100) {
                     await sendSSE(writer: writer, event: "log",
                                  data: "<div class='log-line'>\(line)</div>")
                 }
             }
         }

         private static func runAgentLaunch(

             wineURL: URL, bottleURL: URL, writer: Response.Body.StreamWriter

             // We need the API key

             guard case .anthropic(let apiKey) = provider else {

                              data: "<div class='error'>No Anthropic API key configured. Set ANTHROPIC_API_KEY environment variable.</div>")
                 return

             // Capture writer for the callback closure

                                     .replacingOccurrences(of: "<", with: "&lt;")
                                     .replacingOccurrences(of: ">", with: "&gt;")
                                 await sendSSE(writer: sseWriter, event: "log",
                                              data: "<div class='agent-text'>\(escaped)</div>")
                             case .toolCall(let name):
                                 await sendSSE(writer: sseWriter, event: "tool",
                                              data: "<div class='tool-call'>Tool: \(name)</div>")
                             case .toolResult(let name, let truncated):

                                     .replacingOccurrences(of: "&", with: "&amp;")
                                     .replacingOccurrences(of: "<", with: "&lt;")
                                 await sendSSE(writer: sseWriter, event: "log",
                                              data: "<div class='tool-result'>\(name): \(escaped)</div>")
                             case .cost(_, _, let usd):
                                 await sendSSE(writer: sseWriter, event: "cost",
                                              data: "<div class='cost'>Cost: $\(String(format: "%.4f", usd))</div>")
                             case .budgetWarning(let pct):

                                              data: "<div class='warning'>Budget warning: \(pct)% used</div>")
                             case .status(let msg):
                                 let escaped = msg
                                     .replacingOccurrences(of: "<", with: "&lt;")
                                 await sendSSE(writer: sseWriter, event: "status",
                                              data: "<div>\(escaped)</div>")
                             case .error(let msg):
                                 await sendSSE(writer: sseWriter, event: "error",
                                              data: "<div class='error'>\(msg)</div>")
                             case .completed(let result):
                                 await sendSSE(writer: sseWriter, event: "status",
                                              data: "<div>Agent completed: \(result.iterationsUsed) iterations, $\(String(format: "%.4f", result.estimatedCostUSD))</div>")
                             }
                         }
                     }

                     // Call AIService with the streaming callback.
                     // IMPORTANT: Add an onOutput parameter to AIService.runAgentLoop.
                     // In Sources/cellar/Core/AIService.swift, add a new overload:
                     //
                     // static func runAgentLoop(
                     //     gameId: String, entry: GameEntry, executablePath: String,
                     //     wineURL: URL, bottleURL: URL, wineProcess: WineProcess,
                     //     onOutput: (@Sendable (AgentEvent) -> Void)? = nil
                     // ) -> AIResult<String>
                     //
                     // The body is identical to the existing runAgentLoop but passes
                     // onOutput to the AgentLoop constructor.

                     let aiResult = AIService.runAgentLoop(
                         gameId: gameId,
                         entry: game,
                         executablePath: executablePath,
                         wineURL: wineURL,
                         bottleURL: bottleURL,
                         wineProcess: wineProcess,
                         onOutput: onOutput
                     )
                     continuation.resume(returning: aiResult)
                 }
             }

             switch result {
             case .success(let summary):
                 await sendSSE(writer: writer, event: "status",
                              data: "<div class='success'>Agent finished: \(summary)</div>")
             case .failed(let message):
                 await sendSSE(writer: writer, event: "error",
                              data: "<div class='error'>Agent failed: \(message)</div>")
             case .unavailable:
                 await sendSSE(writer: writer, event: "error",
                              data: "<div class='error'>AI provider unavailable</div>")
             }
         }

         // MARK: - SSE Helpers

         private static func sendSSE(writer: Response.Body.StreamWriter, event: String, data: String) async {
             // SSE format: event: <name>\ndata: <data>\n\n
             // Multi-line data must have each line prefixed with "data: "
             let dataLines = data.components(separatedBy: "\n")
                 .map { "data: \($0)" }
                 .joined(separator: "\n")
             let message = "event: \(event)\n\(dataLines)\n\n"
             let buffer = ByteBuffer(string: message)
             _ = try? await writer.write(.buffer(buffer)).get()
         }

             let gameName: String
