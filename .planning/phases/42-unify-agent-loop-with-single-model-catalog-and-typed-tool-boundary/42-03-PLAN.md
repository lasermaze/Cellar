Enum + metadata shape (per CONTEXT.md decision — string-backed, names only, metadata via dictionary table):

```swift
enum AgentToolName: String, CaseIterable {
    case inspectGame    = "inspect_game"
    case runWineCommand = "run_wine_command"
    case saveSuccess    = "save_success"
    case saveFailure    = "save_failure"
    case updateWiki     = "update_wiki"
    case queryWiki      = "query_wiki"
    // ... ALL existing tool names — see AgentTools.swift toolDefinitions array
}

private struct ToolMetadata {
    let description: String
    let inputSchema: JSONValue
    let pendingAction: ((JSONValue) -> String?)?    // nil for tools with no pending action
}

extension AgentToolName {
    private static let metadata: [AgentToolName: ToolMetadata] = [
        .inspectGame: ToolMetadata(description: "...", inputSchema: ..., pendingAction: nil),
        .runWineCommand: ToolMetadata(
            description: "...",
            inputSchema: ...,
            pendingAction: { input in
                guard case .string(let cmd) = input.objectValue?["command"] else { return nil }
                return "Running: \(cmd)"
            }
        ),
        // ... entries for every case
    ]

    var definition: ToolDefinition {
        let m = Self.metadata[self]!     // table completeness asserted by debug-init check
        return ToolDefinition(name: rawValue, description: m.description, inputSchema: m.inputSchema)
    }

    func pendingActionDescription(for input: JSONValue) -> String? {
        Self.metadata[self]?.pendingAction?(input)
    }
}
```

Pending-action closure is critical: the existing `switch toolName: String` at the old line-692 site formats descriptions using input arguments (e.g., game name, command). RESEARCH Pitfall 5: must be `((JSONValue) -> String?)?` not just `String?`.

Derived `toolDefinitions` (replaces hand-authored array in AgentTools):

```swift
extension AgentTools {
    var toolDefinitions: [ToolDefinition] {
        AgentToolName.allCases.map { $0.definition }
    }
}
```

If multiple call sites read `toolDefinitions` (RESEARCH Open Question #3), the computed property keeps the same shape — no other call sites change.
