Public concrete struct â€” the only type AgentLoop sees:

```swift
struct AgentProvider {
    private let adapter: any ProviderAdapter     // class-bound, mutation safe through ref
    let descriptor: ModelDescriptor

    init(descriptor: ModelDescriptor, apiKey: String, /* other ctor deps from existing providers */ ) {
        self.descriptor = descriptor
        switch descriptor.provider {
        case .anthropic: adapter = AnthropicAdapter(descriptor: descriptor, apiKey: apiKey, ...)
        case .deepseek:  adapter = DeepseekAdapter(descriptor: descriptor, apiKey: apiKey, ...)
        case .kimi:      adapter = KimiAdapter(descriptor: descriptor, apiKey: apiKey, ...)
        }
    }

    var maxOutputTokensLimit: Int { descriptor.maxOutputTokens }
    func pricingPerToken() -> (input: Double, output: Double) {
        (descriptor.inputPricePerToken, descriptor.outputPricePerToken)
    }

    func appendUserMessage(_ content: String)      { adapter.appendUserMessage(content) }
    func appendAssistantResponse(_ text: String, toolCalls: [ToolCall]) { adapter.appendAssistantResponse(text, toolCalls: toolCalls) }
    func appendToolResults(_ results: [ToolResult]) { adapter.appendToolResults(results) }
    func callWithRetry(maxOutputTokens: Int) async throws -> AgentLoopProviderResponse {
        try await adapter.callWithRetry(maxOutputTokens: maxOutputTokens)
    }
}
```

Note: With class-bound adapter (`AnyObject`), the struct's delegating methods are NOT `mutating` â€” the reference is constant, the referent's state mutates internally. This avoids the protocol-existential-mutation pitfall (RESEARCH Pitfall 2).
