Output: Two adapter files. The unified fetchContext path produces equivalent context to today's separate WikiService.fetchContext + CollectiveMemoryService.fetchBestEntry combination — restoring the config-entry capability that was dropped from runAgentLoop (per RESEARCH.md open question #1, the answer chosen here is: restore it).

Plan 01 contracts (already in place):
```swift
protocol KnowledgeStore: Sendable {
    func fetchContext(for gameName: String, environment: EnvironmentFingerprint) async -> String?
    func write(_ entry: KnowledgeEntry) async
    func list(filter: KnowledgeListFilter) async -> [KnowledgeEntryMeta]
}
enum KnowledgeEntry: Codable, Sendable {
    case config(CollectiveMemoryEntry)   // typealias ConfigEntry = CollectiveMemoryEntry
    case gamePage(GamePageEntry)
    case sessionLog(SessionLogEntry)
}
struct KnowledgeCache {
    init(cacheDir: URL, ttl: TimeInterval = 3600)
    func read(key: String) -> Data?
    func readStale(key: String) -> Data?
    func write(key: String, data: Data) throws
    func isFresh(key: String) -> Bool
}
```

    ```swift
    /// Cache-only KnowledgeStore. No network. Reads from ~/.cellar/cache/knowledge/.

    struct KnowledgeStoreLocal: KnowledgeStore {
        let cache: KnowledgeCache

        init(cacheDir: URL = CellarPaths.knowledgeCacheDir) {
            self.cache = KnowledgeCache(cacheDir: cacheDir, ttl: .infinity)  // local never expires
        }

        func fetchContext(for gameName: String, environment: EnvironmentFingerprint) async -> String?
        func write(_ entry: KnowledgeEntry) async
        func list(filter: KnowledgeListFilter) async -> [KnowledgeEntryMeta]
    }
    ```
