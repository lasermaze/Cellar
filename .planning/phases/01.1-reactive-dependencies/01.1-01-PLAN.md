```swift
import Foundation

/// Structured result from a winetricks verb installation.
struct WinetricksResult {
    let verb: String
    let success: Bool
    let timedOut: Bool
    let exitCode: Int32
    let elapsed: TimeInterval
}

/// Reusable service for running winetricks verbs with stale-output timeout protection.
struct WinetricksRunner {
    let winetricksURL: URL
    let wineBinary: URL
    let bottlePath: String  // WINEPREFIX path

    /// Thread-safe timestamp tracker for stale-output detection.
    /// Reuses the NSLock + @unchecked Sendable pattern from WineProcess.StderrCapture.
    private final class OutputMonitor: @unchecked Sendable {
        private var _lastOutputTime = Date()
        private let lock = NSLock()

        func touch() {
            lock.lock()
            _lastOutputTime = Date()
            lock.unlock()
        }

        var lastOutputTime: Date {
            lock.lock()
            defer { lock.unlock() }
            return _lastOutputTime
        }
    }
}
```

**`install(verb:)` method** — the core method, returns `WinetricksResult`:

1. Create `Process()` with `executableURL = winetricksURL`, arguments `["-q", verb]` (unattended mode per decision).
2. Set environment: inherit `ProcessInfo.processInfo.environment`, set `WINEPREFIX = bottlePath`.
3. Set up stdout and stderr `Pipe()` with `readabilityHandler` that:
   - Streams data to `FileHandle.standardOutput` / `FileHandle.standardError` (real-time output)
   - Calls `outputMonitor.touch()` on every non-empty data receipt
4. Record `startTime = Date()`.
5. Start the process with `try process.run()`.
6. **Stale-output detection loop** — instead of `waitUntilExit()`, poll:
   ```swift
   let staleTimeout: TimeInterval = 300  // 5 minutes
   while process.isRunning {
       Thread.sleep(forTimeInterval: 2.0)  // Check every 2 seconds
       let timeSinceOutput = Date().timeIntervalSince(outputMonitor.lastOutputTime)
       if timeSinceOutput > staleTimeout {
           print("\nwinetricks '\(verb)' has produced no output for \(Int(staleTimeout/60)) minutes — killing stalled process.")
           process.terminate()
           // Kill wineserver to clean up any Wine child processes
           let wineProcess = WineProcess(wineBinary: wineBinary, winePrefix: URL(fileURLWithPath: bottlePath))
           try? wineProcess.killWineserver()
           // Brief wait for cleanup
           Thread.sleep(forTimeInterval: 1.0)
           break
       }
   }
   ```
7. **Post-exit drain** — `readDataToEndOfFile()` on both pipes, then set `readabilityHandler = nil` on both.
8. Compute `elapsed = Date().timeIntervalSince(startTime)`.
9. Determine `timedOut` — true if the process was terminated due to stale output (check if `timeSinceOutput > staleTimeout` at the point the loop exited, or track with a local `var didTimeout = false` flag set in the stale branch).
10. Return `WinetricksResult(verb: verb, success: !didTimeout && process.terminationStatus == 0, timedOut: didTimeout, exitCode: process.terminationStatus, elapsed: elapsed)`.

**Key design constraints from CONTEXT.md decisions:**
- Always pass `-q` to winetricks (unattended mode)
- Kill wineserver on timeout to clean up Wine processes
- Reuse NSLock wrapper pattern from WineProcess.StderrCapture for thread safety
- 5-minute stale-output detection (no output for 5 minutes = kill)
